#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
using namespace std;

double s[505];

int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;i++){
            scanf("%lf",&s[i]);
        }
        sort(s,s+n);

        double ans=0;
        if(n%2==0){
            //Å¼Êý
            ans=1.0*(s[n/2]+s[(n/2)-1])/2;
        }else{
            //ÆæÊý
            ans=s[n/2];
        }
        printf("%.3lf\n",ans);
    }
    return 0;
}
