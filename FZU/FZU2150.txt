#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<queue>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long LL;
const int MX=15;
const int INF=1<<30;

int m,n,rear;
char s[MX][MX];
int vis[MX][MX];
int dist[][2]={{-1,0},{0,-1},{1,0},{0,1}};//иовСобср

struct Point{
    int x,y,len;
}P[MX*MX],temp,f;

int BFS(int i,int j){
    memset(vis,0,sizeof(vis));
    queue<Point>work;
    work.push(P[i]);
    work.push(P[j]);

    int nx,ny,cnt=0;
    while(!work.empty()){
        f=work.front();
        work.pop();

        if(vis[f.x][f.y]) continue;

        cnt++;
        vis[f.x][f.y]=1;

        if(cnt==rear){
            return f.len;
        }

        for(int k=0;k<4;k++){
            nx=f.x+dist[k][0];
            ny=f.y+dist[k][1];
            if(nx<1||nx>n||ny<1||ny>m||s[nx][ny]=='.'||vis[nx][ny]) continue;

            temp.x=nx;temp.y=ny;temp.len=f.len+1;
            work.push(temp);
        }
    }
    return -1;
}

int main(){
   // freopen("input.txt","r",stdin);
    int T,ansk=0;
    scanf("%d",&T);
    while(T--){
        rear=0;

        scanf("%d%d",&n,&m);
        for(int i=1;i<=n;i++){
            scanf("%s",s[i]+1);
            for(int j=1;j<=m;j++){
                if(s[i][j]=='#'){
                    temp.x=i;
                    temp.y=j;
                    temp.len=0;
                    P[rear++]=temp;
                }
            }
        }
        if(rear<=2){
            printf("Case %d: 0\n",++ansk);
            continue;
        }

        int ans=INF;
        for(int i=0;i<rear;i++){
            for(int j=i+1;j<rear;j++){
                int t=BFS(i,j);
                if(t!=-1) ans=min(ans,t);
            }
        }

        printf("Case %d: %d\n",++ansk,ans==INF?-1:ans);
    }
    return 0;
}
