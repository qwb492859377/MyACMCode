const int maxn=10000+5;

char FM[]="FM";//¸¸Ç×0,Ä¸Ç×1
int vis[maxn],begin,end;

struct People{
    int f,m;

}p[maxn];

struct LAST{
    int id,dic;
}last[maxn];

bool DFS(int id){
    if(id==end)  return true;
    if(!vis[id]) return false;

    last[p[id].f].id=id;
    last[p[id].f].dic=0;
    if(DFS(p[id].f)) return true;

    last[p[id].m].id=id;
    last[p[id].m].dic=1;
    if(DFS(p[id].m)) return true;

    return false;
}

void print_ans(int id){
    if(last[id].id!=begin){
        print_ans(last[id].id);
    }
    printf("%c",FM[last[id].dic]);
}

            vis[a]=1;
            p[a].f=b;
            p[a].m=c;
        }

        while(m--){
            scanf("%d%d",&a,&b);

            begin=a,end=b;
            if(DFS(begin)){
                printf("1 ");
                print_ans(end);
            }else{
                begin=b,end=a;
                if(DFS(begin)){
                    printf("0 ");
                    print_ans(end);
                }else{
                    printf("Relative");
                }
            }
            printf("\n");
        }
    }
