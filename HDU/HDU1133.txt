#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
const int maxn=200;
int fact[205][maxn];
int res[maxn];
void multiply(int a[],int b){   //大党诵
    int i,temp = 0;
    for(i = 0;i <=maxn-1;i++){
        temp += b * a[i];
        a[i] = temp % 10000;
        temp /= 10000;
    }
}
void Fact(){  //求A乘
    fact[0][0] = fact[1][0] = 1;
    for(int i = 2;i <= 200;i++){
        memcpy(fact[i],fact[i-1],maxn*sizeof(int));
        multiply(fact[i],i);
    }
}
void divide(int a[],int b){       //大党小
    int i,temp = 0;
    for(i =maxn-1;i >=0;i--){
        temp = temp*10000 + a[i];
        a[i] = temp / b;
        temp %= b;
    }
}
void output(int a[]){       //出
    int i = maxn-1;
    while (i >0 && a[i] == 0) i--;
    printf("%d",a[i--]);
    while(i>=0)
    printf("%04d",a[i--]);
    putchar('\n');
}
int main(){
    int m,n,t = 1;
    Fact();
    while(cin >> m >> n , m+n){
        printf("Test #%d:\n",t++);
        if(n > m){
            puts("0");
            continue;
        }
        memcpy(res,fact[m+n],maxn*sizeof(int));
        multiply(res,m-n+1);
        divide(res,m+1);
        output(res);
    }
    return 0;
}
