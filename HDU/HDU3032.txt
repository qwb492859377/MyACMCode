int f(int n) {
    int t = n % 4;
    if(t == 1 || t == 2) {
        return n;
    }
    if(t == 3) return n + 1;
    return n - 1;
}

int main() {
    int T, n; //FIN;
    scanf("%d", &T);
    while(T--) {
        scanf("%d", &n);
        int sum = 0, t;
        for(int i = 1; i <= n; i++) {
            scanf("%d", &t);
            sum ^= f(t);
        }
        printf("%s\n", sum ? "Alice" : "Bob");
    }
    return 0;
}
